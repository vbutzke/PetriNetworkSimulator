
    public int getLargestTransitionNameSize(){

        int size = 0;
        int tSize;

        for (Transition t: transitions) {
            tSize = t.getName().length();
            if(tSize>size)
                size = tSize;
        }

        return size;
    }

    public int getLargestPlaceNameSize(){

        int size = 0;
        int pNameSize;

        for (String p: places.keySet()) {
            pNameSize = p.length();
            if(pNameSize>size)
                size = pNameSize;
        }

        return size;
    }

    public int getLargestPlaceNumSize(){

        int size = 0;
        int pNumSize;

        for (String p: places.keySet()) {
            pNumSize = places.get(p).toString().length();
            if(pNumSize>size)
                size = pNumSize;
        }

        return size;
    }
